796        /**
797         * Change the given times from one timezone to another
798         * 
799         * @param conn
800         *            the current connection to the MySQL server
801         * @param t
802         *            the times to change
803         * @param fromTz
804         *            the timezone to change from
805         * @param toTz
806         *            the timezone to change to
807         * 
808         * @return the times changed to the timezone 'toTz'
809         */
810        public static Time changeTimezone(Connection conn, Time t, TimeZone fromTz,
811                        TimeZone toTz, boolean rollForward) {
812                if ((conn != null) && conn.getUseTimezone()) {
813                        // Convert the timestamp from GMT to the server's timezone
814                        Calendar fromCal = Calendar.getInstance(fromTz);
815                        fromCal.setTime(t);
816 
817                        int fromOffset = fromCal.get(Calendar.ZONE_OFFSET)
818                                        + fromCal.get(Calendar.DST_OFFSET);
819                        Calendar toCal = Calendar.getInstance(toTz);
820                        toCal.setTime(t);
821 
822                        int toOffset = toCal.get(Calendar.ZONE_OFFSET)
823                                        + toCal.get(Calendar.DST_OFFSET);
824                        int offsetDiff = fromOffset - toOffset;
825                        long toTime = toCal.getTime().getTime();
826 
827                        if (rollForward || (conn.isServerTzUTC() && !conn.isClientTzUTC())) {
828                                toTime += offsetDiff;
829                        } else {
830                                toTime -= offsetDiff;
831                        }
832 
833                        Time changedTime = new Time(toTime);
834 
835                        return changedTime;
836                }
837 
838                return t;
839        }
